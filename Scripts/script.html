document.querySelector("form").addEventListener("onmouseover", function (event) { event.preventDefault();// parar envío const nombre = event.target.nombre.value; let validated = true; if (nombre.length < 5) { alert("Nombre muy corto"); validated = false; } // Comprobación final - Todo validado if (validated) { alert("ÉXITO - Formulario correcto y enviado") event.target.submit(); } })
